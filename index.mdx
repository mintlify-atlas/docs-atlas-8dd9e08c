<div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
  <h2 className="text-3xl font-bold text-gray-900 dark:text-white mb-4">Quick Start</h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-8">Get BioAgents running in minutes with these simple steps.</p>
  
  <Steps>
    <Step title="Clone and install dependencies">
      Clone the repository and install dependencies using Bun:
      
      ```bash
      git clone https://github.com/bio-xyz/BioAgents.git
      cd BioAgents
      bun install
      ```
    </Step>
    
    <Step title="Configure environment variables">
      Copy the example environment file and add your API keys:
      
      ```bash
      cp .env.example .env
      ```
      
      Set your LLM provider API keys:
      ```bash
      OPENAI_API_KEY=your_key_here
      ANTHROPIC_API_KEY=your_key_here
      GOOGLE_API_KEY=your_key_here
      ```
    </Step>
    
    <Step title="Set up the database">
      Configure Supabase connection in your `.env`:
      
      ```bash
      SUPABASE_URL=your_supabase_url
      SUPABASE_ANON_KEY=your_anon_key
      SUPABASE_SERVICE_KEY=your_service_key
      ```
    </Step>
    
    <Step title="Start the development server">
      Run the development server:
      
      ```bash
      bun run dev
      ```
      
      The server will start at `http://localhost:3000`
    </Step>
  </Steps>
</div>
